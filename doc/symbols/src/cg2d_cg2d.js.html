 10 /**
 11  * @namespace
 12  * @name cg2d
 13  * @return {object} An interface 
 14  */
 15 var cg2d = (function() {
 16 
 17   var Matrix = lalgebra.Matrix;
 18   var errMsg = lalgebra.errMsg;
 19   var check2dArray = lalgebra.check2dArray;
 20 
 21   /**
 22    * Creates an object.
 23    * 
 24    * @memberof cg2d
 25    * @constructor
 26    * @this {Object2d}
 27    * @param {Array} vertices An array of elements for the Matrix.
 28    * @param {Array} edges An array of edges of the object.
 29    * @param {object} settings An object with properties of segments, points and captions.
 30    */
 31   function Object2d(vertices, edges, settings) {
 32     var cols = 0;
 33     var obj = new Matrix(vertices);
 34 
 35     if(obj.getCols()!==3) {
 36       throw errMsg("Matrix of the two-dimensional object should have three columns!", "cg-ob-001");
 37     }
 38     if(check2dArray(edges)) cols = edges[0].length;
 39 
 40     if(cols!==2) {
 41       throw errMsg("Matrix of edges should have two columns!", "cg-ob-002");
 42     }
 43     if(!(settings["segments"] && settings["points"] && settings["captions"])) {
 44       throw errMsg("Settings should have three properties: segments, points and captions!", "cg-ob-003");
 45     }
 46     /**
 47      * Copies the object.
 48      *
 49      * @memberof cg2d.Object2d
 50      * @return {Object2d} A copy of the object
 51      */
 52     this.getCopy = function() { 
 53       var els = obj.getElems();
 54       return new Object2d(els, edges, settings);
 55       //JSON.parse(JSON.stringify(edges)), JSON.parse(JSON.stringify(settings))
 56     };
 57     /**
 58      * Gets elements of the object matrix.
 59      *
 60      * @memberof cg2d.Object2d
 61      * @return {Array} An array of the object matrix
 62      */
 63     this.getElems = function() { 
 64       return obj.getElems();
 65     };
 66     /**
 67      * Applies a transformation to the object.
 68      *
 69      * @memberof cg2d.Object2d
 70      * @param {Transform2d} tMx Contains a matrix of the transformation.
 71      */
 72     this.applyTransformation = function(tMx) { 
 73       if(tMx instanceof Transform2d) { 
 74         obj = obj.multiply(tMx.getMatrix()).normolize();
 75       }
 76     };
 77     /**
 78      * Displays the array elements into columns like in matrixes.
 79      * @memberof cg2d.Object2d
 80      */
 81     this.trace = function() { obj.trace(); };
 82     /**
 83      * Returns coordinates of the object edges and their visual settings.
 84      *
 85      * @memberof cg2d.Object2d
 86      * @return {object} An object contains coordinates of the edges, their width and color.
 87      */
 88     this.getSegments = function() {
 89       var coords = [];
 90       var tmpArr = obj.getElems();
 91 
 92       var amountEdges = edges.length;
 93       var p1, p2;
 94 
 95       for(var i=0; i<amountEdges; i+=1) {
 96         p1 = tmpArr[edges[i][0]].slice(0, -1);
 97         p2 = tmpArr[edges[i][1]].slice(0, -1);
 98         coords.push({
 99           "p1":{"x":p1[0], "y":p1[1]},  
100           "p2":{"x":p2[0], "y":p2[1]}
101         });
102       }
103       return {
104         "coords": coords,
105         "width": settings.segments.width,
106         "color": settings.segments.color
107       };
108     };
109     /**
110      * Returns coordinates of the object verteces and their visual settings.
111      *
112      * @memberof cg2d.Object2d
113      * @return {object} An object contains coordinates of the verteces, their width and color.
114      */
115     this.getPoints = function() {
116       var coords = [];
117       var tmpArr = obj.getElems();
118       var rows = obj.getRows();
119       var cols = obj.getCols();
120       var p;
121 
122       for(var i=0; i<rows; i+=1) {
123         p = tmpArr[i].slice(0, -1);
124         coords.push({"x":p[0], "y":p[1]});
125       }
126       return {
127         "coords": coords,
128         "width": settings.points.width,
129         "color": settings.points.color
130       };
131     };
132     /**
133      * Returns captions of the verteces and their visual settings.
134      *
135      * @memberof cg2d.Object2d
136      * @return {object} An object contains coordinates of the verteces, their texts, font and color.
137      */
138     this.getCaptions = function() {
139       var coords = this.getPoints().coords;
140 
141       return {
142         "coords": coords,
143         "texts": settings.captions.texts || [],
144         "font": settings.captions.font,
145         "color": settings.captions.color
146       };
147     };
148   }  
149   //==================================================================================
150   /**
151    * Creates a matrix of combined transformations.
152    * 
153    * @memberof cg2d
154    * @constructor
155    * @this {Transform2d}
156    * @param {Array} els An array of elements for the transformation Matrix
157    */
158   function Transform2d(els) {
159     var elements = els ||  [[ 1,0,0 ],
160                             [ 0,1,0 ],
161                             [ 0,0,1 ]];
162     var tMx = new Matrix(elements);
163     /**
164      * Displays the array elements into columns like in matrixes.
165      * @memberof cg2d.Transform2d
166      */
167     this.trace = function() { tMx.trace(); };
168     /**
169      * Returns a copy of the transformation Matrix.
170      *
171      * @memberof cg2d.Transform2d
172      * @return {Matrix} A copy of the transformation Matrix
173      */
174     this.getMatrix = function() { return tMx.getCopy(); };
175     /**
176      * Gets elements of the transformation Matrix.
177      *
178      * @memberof cg2d.Transform2d
179      * @return {Array} An array of the transformation Matrix
180      */
181     this.getElems = function() { return tMx.getElems(); };
182     /**
183      * Multiplies the matrix of combined transformations by shearing matrix.
184      *
185      * @memberof cg2d.Transform2d
186      * @param {string} axis A name of axis.
187      * @param {number} v A value of the transformation.
188      */
189     this.shear = function(axis, v) {
190       var tmpArr; 
191       
192       v = parseFloat(v);
193       if(isNaN(v)) axis = "n"; // function will return a unity martix
194 
195       switch(axis.toLowerCase())
196       {
197         case "x": tmpArr = [[ 1,0,0 ],
198                             [ v,1,0 ],
199                             [ 0,0,1 ]];
200         break;
201         case "y": tmpArr = [[ 1,v,0 ],
202                             [ 0,1,0 ],
203                             [ 0,0,1 ]];
204         break;
205         default:  tmpArr = [[ 1,0,0 ],
206                             [ 0,1,0 ],
207                             [ 0,0,1 ]];
208       }
209       var transform = new Matrix(tmpArr);
210       tMx = tMx.multiply(transform);
211     };    
212     /**
213      * Multiplies the matrix of combined transformations by translation matrix.
214      *
215      * @memberof cg2d.Transform2d
216      * @param {string} axis A name of axis.
217      * @param {number} v A value of the transformation.
218      */
219     this.move = function(axis, v) {
220       var tmpArr; 
221       
222       v = parseFloat(v);
223       if(isNaN(v)) axis = "n"; // function will return a unity martix
224 
225       switch(axis.toLowerCase())
226       {
227         case "x": tmpArr = [[ 1,0,0 ],
228                             [ 0,1,0 ],
229                             [ v,0,1 ]];
230         break;
231         case "y": tmpArr = [[ 1,0,0 ],
232                             [ 0,1,0 ],
233                             [ 0,v,1 ]];
234         break;
235         default:  tmpArr = [[ 1,0,0 ],
236                             [ 0,1,0 ],
237                             [ 0,0,1 ]];
238       }
239       var transform = new Matrix(tmpArr);
240       tMx = tMx.multiply(transform);
241     };    
242     /**
243      * Multiplies the matrix of combined transformations by rotation matrix.
244      *
245      * @memberof cg2d.Transform2d
246      * @param {number} angle An angle of rotation.
247      */
248     this.rotate = function(angle) {
249       var tmpArr; 
250       var a;
251 
252       angle = parseFloat(angle);
253       if(isNaN(angle)) 
254         tmpArr = [[ 1,0,0 ],
255                   [ 0,1,0 ],
256                   [ 0,0,1 ]];
257       else {
258         a = lalgebra.degreeToRadian(angle);
259         tmpArr = [[ Math.cos(a), Math.sin(a), 0],
260                   [-Math.sin(a), Math.cos(a), 0],
261                   [ 0,           0,           1]];
262       }
263       var transform = new Matrix(tmpArr);
264       tMx = tMx.multiply(transform);
265     };
266     /**
267      * Multiplies the matrix of combined transformations by arbitary rotation matrix.
268      *
269      * @memberof cg2d.Transform2d
270      * @param {object} point The arbitary point.
271      * @param {number} angle An angle of rotation.
272      */
273     this.rotateFree = function(point, angle) {
274       var p = {};
275 
276       if(!(point.x && point.y)) {
277         p = { x:0, y:0 };
278       }
279       else {
280         p.x = point.x;
281         p.y = point.y;
282       }
283       angle = parseFloat(angle);
284       if(isNaN(angle)) angle = 0;
285       a = lalgebra.degreeToRadian(angle);
286 
287       this.move("x", -p.x);
288       this.move("y", -p.y);
289       this.rotate(a);
290       this.move("x", p.x);
291       this.move("y", p.y);
292     }
293     /**
294      * Multiplies the matrix of combined transformations by scaling matrix.
295      *
296      * @memberof cg2d.Transform2d
297      * @param {string} axis A name of axis.
298      * @param {number} v A value of the transformation.
299      */
300     this.scale = function(axis, v) {
301       var tmpArr; 
302 
303       v = parseFloat(v);
304       if(isNaN(v)) axis = "n"; // function will return a unity martix
305 
306       switch(axis.toLowerCase())
307       {
308         case "x": tmpArr = [[ v,0,0 ],
309                             [ 0,1,0 ],
310                             [ 0,0,1 ]];
311         break;
312         case "y": tmpArr = [[ 1,0,0 ],
313                             [ 0,v,0 ],
314                             [ 0,0,1 ]];
315         break;
316         case "s": tmpArr = [[ 1,0,0 ],
317                             [ 0,1,0 ],
318                             [ 0,0,v ]];
319         break;
320         default:  tmpArr = [[ 1,0,0 ],
321                             [ 0,1,0 ],
322                             [ 0,0,1 ]];
323      }
324       var transform = new Matrix(tmpArr);
325       tMx = tMx.multiply(transform);
326     };
327   }

329   /**
330    * Creates a scene with three-dimensional objects.
331    * 
332    * @memberof cg2d
333    * @constructor
334    * @this {Scene2d}
335    * @param {CanvasRenderingContext2D} ctx Two-dimensional context.
336    * @param {object} center A center of the canvas.
337    * @param {number} scale A koefficient of scaling.
338    */
339   function Scene2d(ctx, center, scale) {
340     var objects = [];
341     var space = new Transform2d;
342 
343     if(!(ctx instanceof CanvasRenderingContext2D))
344       throw errMsg("Problem with a canvas context!", "cg-sc-001");
345 
346     center = center || {"x":0, "y":0};
347 
348     setSpace();
349     /**
350      * Creates a transformation matrix to convert object coordinates into the screen coordinates.
351      */
352     function setSpace() {
353       space = new Transform2d;
354 
355       space.scale("y", -1);
356       space.scale("s", 1/scale);
357       space.move("x", center.x);
358       space.move("y", center.y);
359     }
360     /**
361      * Removes one or more objects from the object array of the scene.
362      * 
363      * @memberof cg2d.Scene2d
364      * @param {number} idx An index of the object.
365      */
366     this.clearObjects = function(idx) {
367       idx = parseInt(idx);
368       if(isNaN(idx)) objects.splice(0);
369       else objects.splice(idx, 1);
370     }
371     /**
372      * Adds an object to the object array of the scene.
373      * 
374      * @memberof cg2d.Scene2d
375      * @param {Object2d} obj An three-dimensional object.
376      */
377     this.addObject = function(obj) {
378       objects.push(obj);
379     }
380     /**
381      * Displays the objects of the scene.
382      * 
383      * @memberof cg2d.Scene2d
384      */
385     this.draw = function() {
386       var obj, o;
387       ctx.clearRect(0, 0, ctx.canvas.width, ctx.canvas.height);
388       for(var i=0, count=objects.length; i<count; i+=1) {
389         obj = objects[i].getCopy();
390         obj.applyTransformation(space);
391 
392         drawSegments(obj.getSegments());
393         drawPoints(obj.getPoints());
394         drawCaptions(obj.getCaptions());
395       }
396     }
397     /**
398      * Displays the object edges.
399      *
400      * @memberof cg2d.Scene2d
401      */
402     function drawSegments(segments) {
403       var p1, p2;
404       var coords = segments.coords;
405 
406       ctx.beginPath();
407       for(var i=0, count=coords.length; i<count; i+=1) {
408         p1 = coords[i].p1;
409         p2 = coords[i].p2;
410         ctx.moveTo(p1.x, p1.y);
411         ctx.lineTo(p2.x, p2.y);
412       }
413       ctx.strokeStyle = segments.color;
414       ctx.lineWidth = segments.width;
415       ctx.stroke();
416     }
417     /**
418      * Displays the object verteces.
419      *
420      * @memberof cg2d.Scene2d
421      */
422     function drawPoints(points) {
423       var p;
424       var coords = points.coords;
425       var radius = points.width/2;
426 
427       ctx.fillStyle = points.color;
428       for(var i=0, count=coords.length; i<count; i+=1) {
429         p = coords[i];
430         ctx.beginPath();
431         ctx.moveTo(p.x, p.y);
432         ctx.arc(p.x, p.y, radius, 0, Math.PI*2, true);
433         ctx.fill();
434       }
435     }
436     /**
437      * Displays the captions of the verteces.
438      *
439      * @memberof cg2d.Scene2d
440      */
441     function drawCaptions(captions) {
442       var p;
443       var coords = captions.coords;
444       var texts = captions.texts;
445       ctx.font = captions.font;
446       ctx.fillStyle = captions.color;
447       ctx.textBaseline = "top";
448 
449       var count = (texts.length < coords.length)? texts.length : coords.length;
450       for(var i=0; i<count; i+=1) {
451         p = coords[i];
452         ctx.beginPath();
453         ctx.fillText(texts[i], p.x+2, p.y+2);
454         ctx.fill();
455       }
456     }
457   }
458   //==================================================================================
459   return { // An API of cg2d
460     "Object2d": Object2d,
461     "Transform2d": Transform2d,
462     "Scene2d": Scene2d
463   }
